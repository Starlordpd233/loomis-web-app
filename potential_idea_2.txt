                    departmentFilter.appendChild(option);

                const selectedDept = departmentFilter.value;
                const selectedCredits = creditsFilter.value;

                    const matchesSearch = searchTerm === '' ||
                        course.title.toLowerCase().includes(searchTerm) ||
                        course.department.toLowerCase().includes(searchTerm) ||
                        course.description.toLowerCase().includes(searchTerm);
                    const matchesDept = selectedDept === 'all' || course.department === selectedDept;
                    const matchesCredits = selectedCredits === 'all' || course.credits.toString() === selectedCredits;

                    return matchesSearch && matchesDept && matchesCredits;
                });

                renderCourses(filteredCourses);
            };

                let totalCourses = 0;

                    emptyPlanner.classList.remove('hidden');
                    plannerCourseCount.textContent = '0 courses';

                                return `
                                <div draggable="true" data-course-id="${course.id}" data-term="${term}" class="planner-course flex items-center justify-between bg-slate-50 dark:bg-slate-700/50 p-3 rounded-md cursor-grab active:cursor-grabbing transition-shadow duration-300">

                        addCourseToPlanner(courseId, firstTerm);
                    } else {
                        // Handle case where there are no terms
                        const newTerm = "New Term";
                        plannerData[newTerm] = [];
                        addCourseToPlanner(courseId, newTerm);
                    }
                }
            });

            // Planner interaction (using event delegation)
            plannerContainer.addEventListener('click', e => {
                const removeCourseBtn = e.target.closest('.remove-course-btn');
                if(removeCourseBtn) {
                    const { courseId, term } = removeCourseBtn.dataset;
                    plannerData[term] = plannerData[term].filter(id => id !== courseId);
                    renderPlanner();
                    filterAndRender();
                    return;
                }
                
                const removeTermBtn = e.target.closest('.remove-term-btn');
                if(removeTermBtn){
                    const termToRemove = removeTermBtn.dataset.term;
                    delete plannerData[termToRemove];
                    renderPlanner();
                    filterAndRender();
                }
            });

            // Modal closing events
            courseModal.addEventListener('click', e => {
                if (e.target === courseModal) { // Click on backdrop
                    hideCourseModal();
                }
            });
            document.addEventListener('keydown', e => {
                if (e.key === 'Escape' && !courseModal.classList.contains('hidden')) {
                    hideCourseModal();
                }
            });

            // Planner "Add Term" functionality
            newTermInput.addEventListener('keyup', () => {
                addTermBtn.disabled = newTermInput.value.trim() === '';
            });
            addTermBtn.addEventListener('click', () => {
                const termName = newTermInput.value.trim();
                if(termName && !plannerData.hasOwnProperty(termName)){
                    plannerData[termName] = [];
                    renderPlanner();
                    newTermInput.value = '';
                    addTermBtn.disabled = true;
                }
            });

            // Dark Mode
            darkModeToggle.addEventListener('click', () => {
                document.documentElement.classList.toggle('dark');
            });
            // Check for saved dark mode preference
            if (localStorage.theme === 'dark' || (!('theme' in localStorage) && window.matchMedia('(prefers-color-scheme: dark)').matches)) {
                document.documentElement.classList.add('dark')
            } else {
                document.documentElement.classList.remove('dark')
            }
            
            // Drag and Drop functionality
            let draggedItem = null;
            plannerContainer.addEventListener('dragstart', e => {
                const target = e.target.closest('.planner-course');
                if(target) {
                    draggedItem = target;
                    setTimeout(() => {
                        target.classList.add('dragging');
                    }, 0);
                }
            });

            plannerContainer.addEventListener('dragend', e => {
                const target = e.target.closest('.planner-course');
                if(target) {
                   target.classList.remove('dragging');
                   draggedItem = null;
                }
            });

            plannerContainer.addEventListener('dragover', e => {
                e.preventDefault();
                const dropZone = e.target.closest('.drop-zone');
                if (dropZone) {
                    document.querySelectorAll('.drop-zone').forEach(dz => dz.classList.remove('drag-over'));
                    dropZone.classList.add('drag-over');
                }
            });

            plannerContainer.addEventListener('dragleave', e => {
                const dropZone = e.target.closest('.drop-zone');
                if(dropZone){
                    dropZone.classList.remove('drag-over');
                }
            });

            plannerContainer.addEventListener('drop', e => {
                e.preventDefault();
                document.querySelectorAll('.drop-zone').forEach(dz => dz.classList.remove('drag-over'));
                const dropZone = e.target.closest('.drop-zone');
                if (dropZone && draggedItem) {
                    const { courseId, term: oldTerm } = draggedItem.dataset;
                    const newTerm = dropZone.dataset.term;
                    
                    if (oldTerm !== newTerm) {
                        // Remove from old term
                        plannerData[oldTerm] = plannerData[oldTerm].filter(id => id !== courseId);
                        // Add to new term
                        if(!plannerData[newTerm].includes(courseId)) {
                             plannerData[newTerm].push(courseId);
                        }
                    }
                    
                    renderPlanner();
                    filterAndRender();
                }
            });

            // --- INITIALIZATION ---
            populateFilters();
            renderPlanner();
            filterAndRender(); // Initial render
        });
    </script>
</body>
</html>
